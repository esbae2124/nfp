function showLoading(show) {
  document.getElementById('loading-bar').style.display = show ? 'block' : 'none';

function makeDays() {
  return Array.from({length:DAYS}, () => ({type:'none',mucus:'',mfeel:'',emo:'',memo:''}));
}

// ── Google Sheet Sync ────────────────────────────────────
async function loadFromCloud() {
  showLoading(true);
  try {
    // 로컬 캐시 먼저 로드해서 빠르게 보여주기
    const localData = localStorage.getItem('nfp-v2');
    if (localData) {
      cycles = JSON.parse(localData);
      render();
    }
    const localTrash = localStorage.getItem('nfp-v2-trash');
    if (localTrash) {
      trash = JSON.parse(localTrash);
      updateTrashBtn();
    }

    if (SCRIPT_URL.includes("여기에")) {
      // URL 설정 전 초기 샘플 세팅
      if (cycles.length === 0) {
        const today = new Date();
        const start = new Date(today); start.setDate(today.getDate() - 5);
        cycles = [{ startDate: start.toISOString().split('T')[0], days: makeDays() }];
        render();
      }
      return;
    }

    // 클라우드 데이터 가져오기
    const response = await fetch(SCRIPT_URL, {
      method: "POST",
      body: JSON.stringify({ action: "load" })
    });
    const result = await response.json();
    if (result.status === "success" && result.data) {
      cycles = result.data;
      localStorage.setItem('nfp-v2', JSON.stringify(cycles));
      render();
    }
  } catch (e) {
    console.error("클라우드 로드 실패, 로컬 모드로 작동합니다.", e);
  } finally {
    showLoading(false);
  }
}

async function saveToCloud() {
  showLoading(true);
  // 로컬 로리지 선저장 (네트워크 문제 대비)
  localStorage.setItem('nfp-v2', JSON.stringify(cycles));
  localStorage.setItem('nfp-v2-trash', JSON.stringify(trash));
  
  if (SCRIPT_URL.includes("여기에")) {
    showLoading(false);
    return;
  }

  try {
    await fetch(SCRIPT_URL, {
      method: "POST",
      body: JSON.stringify({ action: "save", data: cycles })
    });
  } catch (e) {
    console.error("클라우드 저장 실패", e);
  } finally {
    showLoading(false);
  }
}

  if (trash.length > 0) {
    btn.style.display = '';
    cnt.textContent = trash.length;

  }
}

// ── 피크일 자동 계산 엔진 ──────────────────────────────────
function getComputedDayType(cycle, index) {
  for (let offset = 1; offset <= 3; offset++) {
    const targetIdx = index - offset;
    if (targetIdx >= 0) {
      if (cycle.days[targetIdx].type === 'peak') {
        return `post${offset}`;
      }
    }
  }
  return cycle.days[index].type || 'none';
}

function graphSVG(days) {
  const W = CELL * DAYS;
  const pts = [];
  for (let i = 0; i < DAYS; i++) {
    const k = days[i].emo;
    if (k && EMO[k]) pts.push({x: i*CELL + CELL/2, y: EMO[k].y, color: EMO[k].color});
  }
  let lines = '';
  for (let p = 0; p < pts.length-1; p++) {
    const a = pts[p], b = pts[p+1];
    const mx = (a.x+b.x)/2, my = (a.y+b.y)/2;
    lines += `<line x1="${a.x}" y1="${a.y}" x2="${mx}" y2="${my}" stroke="${a.color}" stroke-width="1.8" stroke-linecap="round"/>`;
    lines += `<line x1="${mx}" y1="${my}" x2="${b.x}" y2="${b.y}" stroke="${b.color}" stroke-width="1.8" stroke-linecap="round"/>`;
  }
  let dots = '';
  for (const p of pts) dots += `<circle cx="${p.x}" cy="${p.y}" r="3" fill="${p.color}"/>`;
  const grid = `
    <line x1="0" y1="8"  x2="${W}" y2="8"  stroke="#eee" stroke-width="0.5" stroke-dasharray="3,3"/>
    <line x1="0" y1="22" x2="${W}" y2="22" stroke="#eee" stroke-width="0.5" stroke-dasharray="3,3"/>
    <line x1="0" y1="36" x2="${W}" y2="36" stroke="#eee" stroke-width="0.5" stroke-dasharray="3,3"/>`;
  return `<svg width="${W}" height="${GH}" viewBox="0 0 ${W} ${GH}" style="display:block;overflow:visible;">${grid}${lines}${dots}</svg>`;
}

// ── Date helpers ─────────────────────────────────────────
function addDays(dateStr, n) {
  const d = new Date(dateStr);
  d.setDate(d.getDate() + n);
  return d;
}
function fmt(dateStr, i) {
  if (!dateStr) return '';
  const d = addDays(dateStr, i);
  return `${d.getMonth()+1}/${d.getDate()}`;
}
function fmtFull(dateStr) {
  if (!dateStr) return '';
  const d = new Date(dateStr);
  return `${d.getFullYear()}년 ${d.getMonth()+1}월 ${d.getDate()}일`;
}
function isToday(dateStr, i) {
  if (!dateStr) return false;
  const t = new Date(); t.setHours(0,0,0,0);
  const d = addDays(dateStr, i); d.setHours(0,0,0,0);
  return d.getTime() === t.getTime();
}
function isFuture(dateStr, i) {
  if (!dateStr) return false;
  const t = new Date(); t.setHours(0,0,0,0);
  const d = addDays(dateStr, i); d.setHours(0,0,0,0);
  return d > t;
}

    html += `</tr>`;

    html += `<tr><td class="rl">날짜</td>`;
    for (let i = 0; i < DAYS; i++) html += `<td class="td-date">${fmt(sd,i)}</td>`;
    html += `</tr>`;

    html += `<tr><td class="rl">기록 표시</td>`;
    for (let i = 0; i < DAYS; i++) {
      const today = isToday(sd, i);
      const future = isFuture(sd, i);
      const cls = today ? ' is-today' : '';
      const computedType = getComputedDayType(cycle, i);
      html += `<td class="td-rec${cls}" onclick="openRec(${ci},${i})">${future ? '' : recSVG(computedType)}</td>`;
    }
    html += `</tr>`;

    html += `<tr><td class="rl">점액 모양</td>`;
    for (let i = 0; i < DAYS; i++) html += `<td class="td-mu" onclick="openRec(${ci},${i})">${mucusShort[cycle.days[i].mucus]||''}</td>`;
    html += `</tr>`;

    html += `<tr><td class="rl">점액 느낌</td>`;
    for (let i = 0; i < DAYS; i++) html += `<td class="td-mu" onclick="openRec(${ci},${i})">${mfeelShort[cycle.days[i].mfeel]||''}</td>`;
    html += `</tr>`;

    html += `<tr><td class="rl emo">감정 그래프</td>`;
    for (let i = 0; i < DAYS; i++) html += `<td class="td-emo" onclick="openRec(${ci},${i})">${emoSVG(cycle.days[i].emo)}</td>`;
    html += `</tr>`;

    html += `<tr>
      <td class="rl emo" style="font-size:10px;">감정선</td>
      <td colspan="${DAYS}" class="td-graph-wrap" style="border-left:0.5px solid var(--border);">${graphSVG(cycle.days)}</td>
    </tr>`;

    html += `<tr><td class="rl">메모</td>`;
    for (let i = 0; i < DAYS; i++) {
      const m = cycle.days[i].memo || '';
      html += `<td class="td-memo" onclick="openRec(${ci},${i})" title="${m}">${m}</td>`;
    }
    html += `</tr>`;

    html += `<tr><td class="rl" style="height:8px; background: transparent; border:none;"></td>${Array(DAYS).fill('<td style="height:8px; border:none;"></td>').join('')}</tr>`;
  });

  document.getElementById('mt').innerHTML = html;

  const types = [
    {type:'period',label:'월경'},{type:'spotting',label:'피비침'},
    {type:'dry',label:'건조'},{type:'mucus',label:'점액관찰'},
    {type:'peak',label:'피크'},{type:'post1',label:'피크+1'},
    {type:'post2',label:'피크+2'},{type:'post3',label:'피크+3'},
  ];
  document.getElementById('legend-row').innerHTML = types.map(t =>
    `<div class="legend-item">${recSVG(t.type)}<span>${t.label}</span></div>`
  ).join('');
}

  document.getElementById('date-overlay').classList.add('open');

  const v = document.getElementById('e-startdate').value;
  if (!v) { alert('날짜를 선택해주세요.'); return; }
  cycles[editDateCi].startDate = v;
  closeDateP(); render(); saveToCloud();
}
function deleteCycle() {
  if (cycles.length === 1) { alert('마지막 주기는 삭제할 수 없습니다.'); return; }
  if (!confirm('이 주기를 삭제하시겠습니까? (휴지통으로 이동)')) return;
  const deleted = cycles.splice(editDateCi, 1)[0];
  deleted._deletedAt = new Date().toISOString();
  trash.push(deleted);
  updateTrashBtn(); closeDateP(); render(); saveToCloud();
}

// ── Trash ────────────────────────────────────────────────
function openTrash() {
  const list = document.getElementById('trash-list');
  const empty = document.getElementById('trash-empty');
  if (trash.length === 0) {
    list.innerHTML = ''; empty.style.display = '';
  } else {
    empty.style.display = 'none';
    list.innerHTML = trash.map((c, i) => {
      const sd = c.startDate ? fmtFull(c.startDate) : '날짜 미설정';
      const delAt = c._deletedAt ? new Date(c._deletedAt).toLocaleDateString('ko-KR') : '';
      return `<div class="trash-item">
        <div>
          <div class="trash-item-info">${sd}</div>
          <div class="trash-item-sub">삭제일: ${delAt}</div>
        </div>
        <div style="display:flex;gap:6px;">
          <button class="btn-restore" onclick="restoreCycle(${i})">복구</button>
          <button class="btn-perm-del" onclick="permDelete(${i})">삭제</button>
        </div>
      </div>`;
    }).join('');
  }
  document.getElementById('trash-overlay').classList.add('open');
  document.getElementById('trash-panel').classList.add('open');
}
function closeTrash() {
  document.getElementById('trash-overlay').classList.remove('open');
  document.getElementById('trash-panel').classList.remove('open');
}
function restoreCycle(i) {
  const c = trash.splice(i, 1)[0];
  delete c._deletedAt;
  cycles.push(c);
  cycles.sort((a,b) => (a.startDate||'').localeCompare(b.startDate||''));
  updateTrashBtn(); closeTrash(); render(); saveToCloud();
}
function permDelete(i) {
  if (!confirm('영구 삭제 시 복구가 불가능합니다. 삭제할까요?')) return;
  trash.splice(i, 1);
  updateTrashBtn(); openTrash(); saveToCloud();
}

// ── Add cycle ────────────────────────────────────────────
function addCycle() {
  const today = new Date().toISOString().split('T')[0];
  cycles.push({ startDate: today, days: makeDays() });
  render(); saveToCloud();
  setTimeout(() => window.scrollTo(0, document.body.scrollHeight), 100);
}
